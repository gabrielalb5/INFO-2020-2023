2) Faça um algoritmo que após a entrada de uma determinada distância entre dois pontos(Km),
e uma determinada velocidade(Km/h), diga qual o tempo médio que levará para chegada
à esse local e qual a velocidade em metros/segundos.

Inicio_algoritmo
	Declare dist, veloc, tempo, veloc_metseg: real;
	
	Escreva("Digite a distancia em Km: ");
	Leia(dist);
	Escreva("Digite a velocidade (Km/h): ");
	Leia(veloc);
	
	tempo <- dist/veloc;
    Escreval("O tempo medio e = ", tempo);
    
    veloc_metseg <- veloc/3.6;
    Escreval("A velocidade em m/s e = ", veloc_metseg);
Fim_algoritmo

Em linguagem C:

#include <stdio.h>

o segundo 30% e o terceiro 20%. Faça um algoritmo que leia o salário de João,
calcule e mostre o valor que cada filho receberá. 

Inicio_algoritmo
	Declare sal, mesada, filho1, filho2, filho3: inteiro
	
	Escreva("Digite o salario: R$");
    Leia(sal);
	
	mesada <- (sal*2)/100;
    
    filho1 <- (mesada*50)/100;
    filho2 <- (mesada*30)/100;

    Escreval("O segundo filho vai receber: R$", filho2);

Fim_algoritmo

Em linguagem C:

4) Maria emprestou dinheiro de sua irmã mais velha. Ela irá pagar em 3 meses. Em cada mês o valor
devido será aumentado com juros de 1%. Faça um algoritmo que leia o valor emprestado,
calcule e mostre o total devido em cada mês subsequente, leia o valor pago no primeiro mês
e calcule novamente o valor devido, e assim por diante. Qual será o valor a pagar no último mês?
Qual foi o total pago? Quanto a mais Maria teve que pagar pelo empréstimo?

Inicio_algoritmo
	Declare dinheiro, parcela, mes1, mes2, mes3, total, pagou_a_mais: real;
	
	Escreva("Digite o valor do emprestimo: R$");
	Leia(dinheiro);
	
	parcela <- dinheiro/3;
    mes1 <- parcela+(parcela*1)/100;
    mes2 <- mes1+(mes1*1)/100;
    mes3 <- mes2+(mes2*1)/100;
	
	total <- mes1+mes2+mes3;
    pagou_a_mais <- (mes1+mes2+mes3)-(dinheiro);
	
	Escreval("Maria pagou R$",mes3," no ultimo mes");
    Escreval("O total pago foi de R$", total);
    Escreval("Ela pagou R$",pagou_a_mais," a mais.");
Fim_algoritmo
	
Em linguagem C:

#include <stdio.h>
int main(){
    
    float dinheiro, parcela, mes1, mes2, mes3, total, pagou_a_mais;

    printf("Digite o valor do emprestimo: R$");
    scanf("%f",&dinheiro);
    
    parcela = dinheiro/3;
    mes1 = parcela+(parcela*1)/100;
    mes2 = mes1+(mes1*1)/100;
    mes3 = mes2+(mes2*1)/100;
    
    total = mes1+mes2+mes3;
    pagou_a_mais = (mes1+mes2+mes3)-(dinheiro);
    
    printf("Maria pagou R$%.2f no ultimo mes\n", mes3);
    printf("O total pago foi de R$%.2f\n", total);
    printf("Ela pagou R$%.2f a mais.", pagou_a_mais);
    
    return 0;
}
